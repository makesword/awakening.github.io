0x01 利用MySQL内置变量获取敏感信息

1.select version(); --获取数据库版本信息
2.select user(); --获取数据库的用户
3.select database(); --获取当前数据库名
4.select @@datadir; --获取当前数据库路径
5.select @@version_compile_os; --操作系统版本
